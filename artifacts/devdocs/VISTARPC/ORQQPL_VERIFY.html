 N NOW,CHNGE

 I $P(^AUPNPROB(GMPIFN,1),U,2)'="T" D  Q  ; BAIL OUT - ALREADY VERIFIED

 . S RETURN(0)="Record in use. Try again in a few moments"

 D AUDIT^GMPLX(CHNGE,"")
 D DTMOD^GMPLX(GMPIFN)

 S RETURN=1
